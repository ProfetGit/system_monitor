28static void format_bytes(unsigned long bytes, char *buf, size_t size) {
29 const char *units[] = {"B", "KB", "MB", "GB", "TB"};
30 int i = 0;
31 double size_d = bytes;
32
33 while (size_d >= 1024 && i < 4) {
34 size_d /= 1024;
35 i++;
36 }
37
38 snprintf(buf, size, "%.1f %s", size_d, units[i]);
39}

41int init_display(void) {
42 // Initialize ncurses
43 if (!initscr()) {
44 fprintf(stderr, "Failed to initialize ncurses\n");
45 return -1;
46 }
47
48 // Check terminal size
49 if (LINES < (CPU_WIN_HEIGHT + MEM_WIN_HEIGHT + DISK_WIN_HEIGHT + GPU_WIN_HEIGHT + 4) ||
50 COLS < (WIN_WIDTH + 2)) {
51 endwin();
52 fprintf(stderr, "Terminal window too small. Minimum size required: %dx%d\n",
53 WIN_WIDTH + 2, CPU_WIN_HEIGHT + MEM_WIN_HEIGHT + DISK_WIN_HEIGHT + GPU_WIN_HEIGHT + 4);
54 return -1;
55 }
56
57 // Set up colors if supported
58 if (has_colors()) {
59 start_color();
60 init_pair(1, COLOR_GREEN, COLOR_BLACK);
61 init_pair(2, COLOR_YELLOW, COLOR_BLACK);
62 init_pair(3, COLOR_RED, COLOR_BLACK);
63 init_pair(4, COLOR_CYAN, COLOR_BLACK);
64 }
65
66 // Configure ncurses
67 cbreak(); // Line buffering disabled
68 noecho(); // Don't echo keystrokes
69 curs_set(0); // Hide cursor if possible
70 timeout(0); // Non-blocking input
71 keypad(stdscr, TRUE); // Enable keypad input
72
73 // Calculate window positions
74 int start_y = (LINES - (CPU_WIN_HEIGHT + MEM_WIN_HEIGHT + DISK_WIN_HEIGHT + GPU_WIN_HEIGHT + 3)) / 2;
75 int start_x = (COLS - WIN_WIDTH) / 2;
76
77 // Create windows with error checking
78 cpu_win = newwin(CPU_WIN_HEIGHT, WIN_WIDTH, start_y, start_x);
79 if (!cpu_win) {
80 endwin();
81 fprintf(stderr, "Failed to create CPU window\n");
82 return -1;
83 }
84
85 mem_win = newwin(MEM_WIN_HEIGHT, WIN_WIDTH,
86 start_y + CPU_WIN_HEIGHT + 1, start_x);
87 if (!mem_win) {
88 delwin(cpu_win);
89 endwin();
90 fprintf(stderr, "Failed to create Memory window\n");
91 return -1;
92 }
93
94 disk_win = newwin(DISK_WIN_HEIGHT, WIN_WIDTH,
95 start_y + CPU_WIN_HEIGHT + MEM_WIN_HEIGHT + 2, start_x);
96 if (!disk_win) {
97 delwin(cpu_win);
98 delwin(mem_win);
99 endwin();
100 fprintf(stderr, "Failed to create Disk window\n");
101 return -1;
102 }
103
104 gpu_win = newwin(GPU_WIN_HEIGHT, WIN_WIDTH,
105 start_y + CPU_WIN_HEIGHT + MEM_WIN_HEIGHT + DISK_WIN_HEIGHT + 3, start_x);
106 if (!gpu_win) {
107 delwin(cpu_win);
108 delwin(mem_win);
109 delwin(disk_win);
110 endwin();
111 fprintf(stderr, "Failed to create GPU window\n");
112 return -1;
113 }
114
115 // Enable scrolling for disk and GPU windows
116 scrollok(disk_win, TRUE);
117 scrollok(gpu_win, TRUE);
118
119 // Clear the screen and refresh
120 clear();
121 refresh();
122
123 return 0;
124}

190 format_bytes(stats->memory.cached, buf, sizeof(buf));
191 mvwprintw(mem_win, 3, 25, "Cached: %s", buf);
192
193 // Swap memory
194 if (stats->memory.swap_total > 0) {
195 color = get_usage_color(stats->memory.swap_usage);
196 if (color) wattron(mem_win, COLOR_PAIR(color));
197 mvwprintw(mem_win, 4, 2, "Swap: %.1f%% used", stats->memory.swap_usage);
198 if (color) wattroff(mem_win, COLOR_PAIR(color));
199 format_bytes(stats->memory.swap_total - stats->memory.swap_free, buf, sizeof(buf));

300
301 // Display power usage if available
302 if (gpu->power_usage > 0) {
303 mvwprintw(gpu_win, y, 4, "Power Usage: %.1f W", gpu->power_usage / 1000.0);
304 y++;
305 }

308 y++;
309 }
310
311 // Add a separator line between GPUs
312 if (i + 1 < stats->gpus.count && y < GPU_WIN_HEIGHT - 1) {
313 mvwhline(gpu_win, y, 1, ACS_HLINE, WIN_WIDTH - 2);
314 y++;
315 }
316 }
317 } else {
318 mvwprintw(gpu_win, 0, 2, " GPU Information (No GPUs detected) ");
319 mvwprintw(gpu_win, 1, 2, "No GPU information available");

322 wrefresh(gpu_win);
323
324 // Refresh the screen
325 refresh();
326}
