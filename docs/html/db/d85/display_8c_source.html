41static void format_bytes(unsigned long bytes, char *buf, size_t size) {
42 const char *units[] = {"B", "KB", "MB", "GB", "TB"};
43 int i = 0;
44 double size_d = bytes;
45
46 while (size_d >= 1024 && i < 4) {
47 size_d /= 1024;
48 i++;
49 }
50
51 snprintf(buf, size, "%.1f %s", size_d, units[i]);

102int init_display(void) {
103 // Initialize ncurses
104 if (!initscr()) {
105 fprintf(stderr, "Failed to initialize ncurses\n");
106 return -1;
107 }
108
109 // Check terminal size
110 int required_height = HEADER_HEIGHT + CPU_WIN_HEIGHT + MEM_WIN_HEIGHT +
111 DISK_WIN_HEIGHT + NET_WIN_HEIGHT + GPU_WIN_HEIGHT +
112 PADDING * 6;
113
114 if (LINES < required_height || COLS < (WIN_WIDTH + 2)) {
115 endwin();
116 fprintf(stderr, "Terminal too small. Minimum size: %dx%d\n",
117 WIN_WIDTH + 2, required_height);
118 return -1;
119 }
120
121 // Setup colors
122 start_color();
123 use_default_colors();
124 init_pair(COLOR_HEADER, COLOR_CYAN, -1);
125 init_pair(COLOR_NORMAL, -1, -1);
126 init_pair(COLOR_WARNING, COLOR_YELLOW, -1);
127 init_pair(COLOR_CRITICAL, COLOR_RED, -1);
128 init_pair(COLOR_GOOD, COLOR_GREEN, -1);
129 init_pair(COLOR_BORDER, COLOR_BLUE, -1);
130
131 // Enable keyboard input and function keys
132 keypad(stdscr, TRUE);
133 noecho();
134 curs_set(0);
135 timeout(0);
136
137 // Reduce screen flicker
138 nodelay(stdscr, TRUE); // Non-blocking input
139 leaveok(stdscr, TRUE); // Don't care where cursor is left
140 cbreak(); // Disable line buffering
141 noecho(); // Don't echo input
142 curs_set(0); // Hide cursor
143
144 // Calculate window positions
145 int current_y = 0;
146 header_win = create_centered_win(HEADER_HEIGHT, WIN_WIDTH, current_y);
147 current_y += HEADER_HEIGHT + PADDING;
148
149 cpu_win = create_centered_win(CPU_WIN_HEIGHT, WIN_WIDTH, current_y);
150 current_y += CPU_WIN_HEIGHT + PADDING;
151
152 mem_win = create_centered_win(MEM_WIN_HEIGHT, WIN_WIDTH, current_y);
153 current_y += MEM_WIN_HEIGHT + PADDING;
154
155 disk_win = create_centered_win(DISK_WIN_HEIGHT, WIN_WIDTH, current_y);
156 current_y += DISK_WIN_HEIGHT + PADDING;
157
158 net_win = create_centered_win(NET_WIN_HEIGHT, WIN_WIDTH, current_y);
159 current_y += NET_WIN_HEIGHT + PADDING;
160
161 gpu_win = create_centered_win(GPU_WIN_HEIGHT, WIN_WIDTH, current_y);
162
163 if (!header_win || !cpu_win || !mem_win || !disk_win || !net_win || !gpu_win) {
164 cleanup_display();
165 return -1;
166 }
167
168 // Enable scrolling and optimize window updates
169 scrollok(stdscr, FALSE);
170
171 // Optimize all windows
172 WINDOW *windows[] = {header_win, cpu_win, mem_win, disk_win, net_win, gpu_win};
173 for (int i = 0; i < 6; i++) {
174 scrollok(windows[i], FALSE);
175 leaveok(windows[i], TRUE);
176 idlok(windows[i], TRUE);
177 idcok(windows[i], TRUE);
178 }
179
180 // Draw initial window borders
181 draw_fancy_box(header_win, "");
182 draw_fancy_box(cpu_win, "CPU");
183 draw_fancy_box(mem_win, "Memory");
184 draw_fancy_box(disk_win, "Disk");
185 draw_fancy_box(net_win, "Network");
186 draw_fancy_box(gpu_win, "GPU");
187
188 refresh();
189 return 0;
190}

206void display_stats(const SystemStats *stats) {
207 char buf[256];
208 int row;
209
210 // Header - only update the time
211 mvwprintw(header_win, 1, (WIN_WIDTH - 14) / 2, "SYSTEM MONITOR");
212 wrefresh(header_win);
213
214 // CPU
215 werase(cpu_win);
216 row = 1;
217 mvwprintw(cpu_win, row++, 2, "CPU Usage: ");
218 wattron(cpu_win, A_BOLD);
219 wprintw(cpu_win, "%.1f%%", stats->cpu.usage);
220 wattroff(cpu_win, A_BOLD);
221 mvwprintw(cpu_win, row++, 2, "Model: %s", stats->cpu.model_name);
222 mvwprintw(cpu_win, row++, 2, "Cores: %u", stats->cpu.cores);
223 draw_fancy_box(cpu_win, "CPU");
224 wrefresh(cpu_win);
225
226 // Memory
227 werase(mem_win);
228 row = 1;
229 format_bytes(stats->memory.total, buf, sizeof(buf));
230 mvwprintw(mem_win, row++, 2, "Total Memory: %s", buf);
231 format_bytes(stats->memory.used, buf, sizeof(buf));
232 mvwprintw(mem_win, row++, 2, "Used Memory: %s (%.1f%%)",
233 buf, stats->memory.usage);
234 format_bytes(stats->memory.free, buf, sizeof(buf));
235 mvwprintw(mem_win, row++, 2, "Free Memory: %s", buf);
236 format_bytes(stats->memory.cached, buf, sizeof(buf));
237 mvwprintw(mem_win, row++, 2, "Cache: %s", buf);
238 mvwprintw(mem_win, row++, 2, "Swap Usage: %.1f%%", stats->memory.swap_usage);
239 draw_fancy_box(mem_win, "Memory");
240 wrefresh(mem_win);
241
242 // Network
243 werase(net_win);
244 row = 1;
245 for (int i = 0; i < stats->network.interface_count && i < 3; i++) {
246 const NetworkInterfaceStats *if_stats = &stats->network.interfaces[i];
247 mvwprintw(net_win, row++, 2, "Interface: %s", if_stats->interface);
248
249 format_speed(if_stats->receive_speed, buf, sizeof(buf));
250 mvwprintw(net_win, row++, 4, "RX: %s", buf);
251
252 format_speed(if_stats->send_speed, buf, sizeof(buf));
253 mvwprintw(net_win, row++, 4, "TX: %s", buf);
254 row++;
255 }
256 draw_fancy_box(net_win, "Network");
257 wrefresh(net_win);
258
259 // Disk
260 werase(disk_win);
261 row = 1;
262 mvwprintw(disk_win, row++, 2, "Disk Usage:");
263 for (int i = 0; i < stats->disks.count && i < 3; i++) {
264 const DiskStats *disk = &stats->disks.disks[i];
265 mvwprintw(disk_win, row++, 2, " %s: %.1f%% used",
266 disk->mount_point,
267 disk->usage);
268 format_bytes(disk->total, buf, sizeof(buf));
269 mvwprintw(disk_win, row++, 4, "Total: %s", buf);
270 format_bytes(disk->available, buf, sizeof(buf));
271 mvwprintw(disk_win, row++, 4, "Free: %s", buf);
272 }
273 draw_fancy_box(disk_win, "Disk");
274 wrefresh(disk_win);
275
276 // GPU
277 werase(gpu_win);
278 row = 1;
279 for (unsigned int i = 0; i < stats->gpus.count && i < 2; i++) {
280 const GPUStats *gpu = &stats->gpus.gpus[i];
281 mvwprintw(gpu_win, row++, 2, "GPU %u: %s", i, gpu->name);
282 mvwprintw(gpu_win, row++, 4, "Usage: %.1f%%", gpu->utilization);
283 mvwprintw(gpu_win, row++, 4, "Temperature: %d°C", gpu->temperature);
284 if (gpu->memory_total > 0) {
285 format_bytes(gpu->memory_used, buf, sizeof(buf));
286 mvwprintw(gpu_win, row++, 4, "Memory Used: %s", buf);
287 }
288 row++;
289 }
290 draw_fancy_box(gpu_win, "GPU");
291 wrefresh(gpu_win);
292
293 // Use doupdate() instead of refresh() for smoother updates
294 doupdate();
295}
