40static int get_disk_io_stats(const char *device, unsigned long *reads,
41 unsigned long *writes, unsigned long *io_in_progress) {
42 FILE *fp = fopen("/proc/diskstats", "r");
43 if (!fp) return -1;
44
45 char line[256];
46 char dev_name[64];
47 unsigned long rd_ios, rd_merges, rd_sectors, rd_ticks;
48 unsigned long wr_ios, wr_merges, wr_sectors, wr_ticks;
49 unsigned long ios_pgr, tot_ticks, rq_ticks;
50
51 // Extract the base device name (e.g., "sda" from "/dev/sda1")
52 const char *base_dev = strrchr(device, '/');
53 if (base_dev) {
54 base_dev++;
55 } else {
56 base_dev = device;
57 }
58
59 // Remove partition numbers for matching
60 char base_name[32];
61 strncpy(base_name, base_dev, sizeof(base_name) - 1);
62 base_name[sizeof(base_name) - 1] = '\0';
63 char *p = base_name;
64 while (*p) {
65 if (isdigit(*p)) {
66 *p = '\0';
67 break;
68 }
69 p++;
70 }
71
72 while (fgets(line, sizeof(line), fp)) {
73 if (sscanf(line, "%*d %*d %s %lu %lu %lu %lu %lu %lu %lu %lu %lu %lu %lu",
74 dev_name,
75 &rd_ios, &rd_merges, &rd_sectors, &rd_ticks,
76 &wr_ios, &wr_merges, &wr_sectors, &wr_ticks,
77 &ios_pgr, &tot_ticks, &rq_ticks) == 12) {
78 if (strcmp(dev_name, base_name) == 0) {
79 *reads = rd_ios;
80 *writes = wr_ios;
81 *io_in_progress = ios_pgr;
82 fclose(fp);
83 return 0;

98int update_disk_stats(DiskInfo *info) {
99 if (!info) return -1;
100
101 FILE *mtab = setmntent("/etc/mtab", "r");
102 if (!mtab) return -1;
103
104 struct mntent *ent;
105 info->count = 0;
106
107 while ((ent = getmntent(mtab)) && info->count < MAX_DISKS) {
108 // Skip non-disk filesystems
109 if (!is_real_disk(ent->mnt_fsname)) continue;
110
111 DiskStats *disk = &info->disks[info->count];
112 struct statvfs fs_stats;
113
114 // Get filesystem statistics
115 if (statvfs(ent->mnt_dir, &fs_stats) == 0) {
116 strncpy(disk->device, ent->mnt_fsname, MAX_DISK_NAME - 1);
117 disk->device[MAX_DISK_NAME - 1] = '\0';
118
119 strncpy(disk->mount_point, ent->mnt_dir, MAX_MOUNT_PATH - 1);
120 disk->mount_point[MAX_MOUNT_PATH - 1] = '\0';
121
122 disk->total = fs_stats.f_blocks * fs_stats.f_frsize;
123 disk->free = fs_stats.f_bfree * fs_stats.f_frsize;
124 disk->available = fs_stats.f_bavail * fs_stats.f_frsize;
125
126 if (disk->total > 0) {
127 disk->usage = 100.0 * (1.0 - ((double)disk->available / disk->total));
128 } else {
129 disk->usage = 0.0;
130 }
131
132 // Get I/O statistics
133 unsigned long reads, writes, io_in_progress;
134 if (get_disk_io_stats(disk->device, &reads, &writes, &io_in_progress) == 0) {
135 disk->reads = reads - prev_reads[info->count];
136 disk->writes = writes - prev_writes[info->count];
137 disk->io_in_progress = io_in_progress;
138
139 prev_reads[info->count] = reads;
140 prev_writes[info->count] = writes;
141 } else {
142 disk->reads = 0;
143 disk->writes = 0;
144 disk->io_in_progress = 0;
145 }
146
147 info->count++;
148 }
149 }
150
151 endmntent(mtab);
152 return 0;
153}
