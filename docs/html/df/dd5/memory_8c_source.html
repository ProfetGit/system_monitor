40static int read_proc_meminfo(MemoryStats *stats) {
41 FILE *fp = fopen("/proc/meminfo", "r");
42 if (!fp) return -1;
43
44 char line[256];
45 unsigned long memTotal = 0, memFree = 0, memAvailable = 0;
46 unsigned long buffers = 0, cached = 0, swapTotal = 0, swapFree = 0;
47 unsigned long shmem = 0, sreclaimable = 0;
48
49 while (fgets(line, sizeof(line), fp)) {
50 unsigned long value;
51 if (sscanf(line, "MemTotal: %lu kB", &value) == 1)
52 memTotal = value * KB_TO_BYTES;
53 else if (sscanf(line, "MemFree: %lu kB", &value) == 1)
54 memFree = value * KB_TO_BYTES;
55 else if (sscanf(line, "MemAvailable: %lu kB", &value) == 1)
56 memAvailable = value * KB_TO_BYTES;
57 else if (sscanf(line, "Buffers: %lu kB", &value) == 1)
58 buffers = value * KB_TO_BYTES;
59 else if (sscanf(line, "Cached: %lu kB", &value) == 1)
60 cached = value * KB_TO_BYTES;
61 else if (sscanf(line, "SwapTotal: %lu kB", &value) == 1)
62 swapTotal = value * KB_TO_BYTES;
63 else if (sscanf(line, "SwapFree: %lu kB", &value) == 1)
64 swapFree = value * KB_TO_BYTES;
65 else if (sscanf(line, "Shmem: %lu kB", &value) == 1)
66 shmem = value * KB_TO_BYTES;
67 else if (sscanf(line, "SReclaimable: %lu kB", &value) == 1)
68 sreclaimable = value * KB_TO_BYTES;
69 }
70
71 fclose(fp);
72
73 if (memTotal == 0) return -1; // Failed to read memory info
74
75 // Calculate actual memory values
76 stats->total = memTotal;
77 stats->free = memFree;
78 stats->available = memAvailable;
79 stats->buffers = buffers;
80 stats->cached = cached + sreclaimable - shmem; // Actual cache
81 stats->swap_total = swapTotal;
82 stats->swap_free = swapFree;
83
84 // Calculate used memory (excluding cache and buffers)
85 unsigned long used = memTotal - memFree - buffers - (cached + sreclaimable - shmem);
86 stats->used = used;
87
88 // Calculate usage percentages
89 stats->usage = 100.0 * ((double)used / memTotal);
90 if (swapTotal > 0) {
91 stats->swap_usage = 100.0 * (1.0 - ((double)swapFree / swapTotal));
92 } else {
93 stats->swap_usage = 0.0;
94 }
95
96 return 0;
97}
