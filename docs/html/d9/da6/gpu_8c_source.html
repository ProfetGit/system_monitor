33static int load_nvml(void) {

65static void read_sysfs_gpu_info(GPUInfo *info) {
66 FILE *fp;
67 char path[256];
68 char line[256];
69 unsigned int gpu_count = 0;
70
71 // Try to find GPUs in sysfs
72 for (unsigned int i = 0; i < MAX_GPUS; i++) {
73 snprintf(path, sizeof(path), "/sys/class/drm/card%u/device/vendor", i);
74 fp = fopen(path, "r");
75 if (!fp) continue;
76
77 GPUStats *gpu = &info->gpus[gpu_count];
78 gpu->supported = 0; // Limited support for non-NVIDIA GPUs
79
80 // Try to read GPU name
81 snprintf(path, sizeof(path), "/sys/class/drm/card%u/device/product", i);
82 FILE *name_fp = fopen(path, "r");
83 if (name_fp) {
84 if (fgets(line, sizeof(line), name_fp)) {
85 line[strcspn(line, "\n")] = 0;
86 strncpy(gpu->name, line, MAX_GPU_NAME - 1);
87 gpu->name[MAX_GPU_NAME - 1] = '\0';
88 }
89 fclose(name_fp);
90 } else {
91 strcpy(gpu->name, "Unknown GPU");
92 }
93
94 gpu_count++;
95 fclose(fp);
96
97 if (gpu_count >= MAX_GPUS) break;
98 }
99
100 info->count = gpu_count;
101}

113int update_gpu_stats(GPUInfo *info) {
114 if (!info) return -1;
115
116 info->count = 0;
117 info->nvidia_available = 0;
118
119 // Try NVIDIA GPUs first
120 if (nvml.handle) {
121 unsigned int device_count = 0;
122 if (nvml.nvmlDeviceGetCount(&device_count) == 0) {
123 info->nvidia_available = 1;
124 info->count = (device_count > MAX_GPUS) ? MAX_GPUS : device_count;
125
126 for (unsigned int i = 0; i < info->count; i++) {
127 void *device_handle = NULL;
128 GPUStats *gpu = &info->gpus[i];
129 gpu->supported = 1;
130
131 if (nvml.nvmlDeviceGetHandleByIndex(i, &device_handle) == 0) {
132 // Get GPU name
133 nvml.nvmlDeviceGetName(device_handle, gpu->name, MAX_GPU_NAME);
134
135 // Get temperature
136 unsigned int temp;
137 if (nvml.nvmlDeviceGetTemperature(device_handle, 0, &temp) == 0) {
138 gpu->temperature = (int)temp;
139 }
140
141 // Get utilization
142 struct {
143 unsigned int gpu;
144 unsigned int memory;
145 } utilization = {0};
146 if (nvml.nvmlDeviceGetUtilizationRates(device_handle, &utilization) == 0) {
147 gpu->utilization = utilization.gpu;
148 }
149
150 // Get memory info
151 struct {
152 unsigned long long total;
153 unsigned long long free;
154 unsigned long long used;
155 } memory = {0};
156 if (nvml.nvmlDeviceGetMemoryInfo(device_handle, &memory) == 0) {
157 gpu->memory_total = memory.total;
158 gpu->memory_free = memory.free;
159 gpu->memory_used = memory.used;
160 }
161
162 // Get power usage
163 unsigned int power;
164 if (nvml.nvmlDeviceGetPowerUsage(device_handle, &power) == 0) {
165 gpu->power_usage = (int)power;
166 }
167
168 // Get fan speed
169 unsigned int fan;
170 if (nvml.nvmlDeviceGetFanSpeed(device_handle, &fan) == 0) {
171 gpu->fan_speed = (int)fan;
172 }
173 }
174 }
175 return 0;
176 }
177 }
178
179 // Fall back to sysfs for non-NVIDIA GPUs
180 read_sysfs_gpu_info(info);
181 return 0;
182}
