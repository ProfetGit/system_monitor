75static void calculate_speeds(NetworkInterfaceStats *current, PreviousStats *prev) {
76 struct timespec now;
77 clock_gettime(CLOCK_MONOTONIC, &now);
78
79 double time_diff = (now.tv_sec - prev->timestamp.tv_sec) +
80 (now.tv_nsec - prev->timestamp.tv_nsec) / 1e9;
81
82 if (time_diff > 0) {
83 current->receive_speed = (current->bytes_received - prev->bytes_received) / time_diff;
84 current->send_speed = (current->bytes_sent - prev->bytes_sent) / time_diff;
85 }
86
87 // Update previous stats
88 prev->bytes_received = current->bytes_received;
89 prev->bytes_sent = current->bytes_sent;
90 prev->timestamp = now;
91}

98int update_network_stats(NetworkStats *stats) {

101 FILE *fp = fopen(PROC_NET_DEV, "r");
102 if (!fp) return -1;
103
104 char line[LINE_BUF_SIZE];
105 int interface_count = 0;
106
107 // Skip header lines
108 fgets(line, sizeof(line), fp);
109 fgets(line, sizeof(line), fp);
110
111 // Read interface statistics
112 while (fgets(line, sizeof(line), fp) && interface_count < MAX_INTERFACES) {
113 if (parse_interface_line(line, &stats->interfaces[interface_count]) == 0) {
114 calculate_speeds(&stats->interfaces[interface_count],
115 &previous_stats[interface_count]);
116 interface_count++;
117 }
118 }
119
120 stats->interface_count = interface_count;
